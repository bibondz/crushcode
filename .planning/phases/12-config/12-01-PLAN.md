```zig
pub const CompletedWork = struct {
    task_id: []const u8,  // owned — dupe from source
    success: bool,
    output: []const u8,   // owned — dupe from source
    duration_ms: u64,
};

pub const CompletionQueue = struct {
    mutex: std.Thread.Mutex,
    items: array_list_compat.ArrayList(CompletedWork),

    pub fn init() CompletionQueue {
        return .{
            .mutex = std.Thread.Mutex{},
            .items = array_list_compat.ArrayList(CompletedWork).init(undefined), // set allocator on first use
        };
    }

    pub fn push(self: *CompletionQueue, allocator: Allocator, item: CompletedWork) !void {
        self.mutex.lock();
        defer self.mutex.unlock();
        // Initialize if needed
        if (self.items.items.len == 0 and self.items.capacity == 0) {
            self.items = array_list_compat.ArrayList(CompletedWork).init(allocator);
        }
        try self.items.append(item);
    }

    pub fn drain(self: *CompletionQueue) []CompletedWork {
        self.mutex.lock();
        defer self.mutex.unlock();
        const result = self.items.items;
        // Reset without freeing — caller takes ownership
        self.items = array_list_compat.ArrayList(CompletedWork).init(self.items.allocator);
        return result;
    }

    pub fn deinit(self: *CompletionQueue) void {
        self.mutex.lock();
        for (self.items.items) |*item| {
            // Free owned strings
        }
        self.items.deinit();
        self.mutex.unlock();
    }
};
```

Actually, simplify: The CompletionQueue holds the allocator directly, initialized once. Push allocates owned copies. Drain returns a snapshot and clears internal list (caller frees the returned items).

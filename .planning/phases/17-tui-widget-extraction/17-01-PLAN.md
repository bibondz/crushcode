## Widget Pattern (vaxis vxfw)

Every widget follows this exact pattern:
```zig
const SomeWidget = struct {
    // Data fields — NO *Model reference
    field1: Type1,
    field2: Type2,

    fn widget(self: *const SomeWidget) vxfw.Widget {
        return .{
            .userdata = @constCast(self),
            .drawFn = typeErasedDrawFn,
        };
    }

    fn typeErasedDrawFn(ptr: *anyopaque, ctx: vxfw.DrawContext) std.mem.Allocator.Error!vxfw.Surface {
        const self: *const SomeWidget = @ptrCast(@alignCast(ptr));
        return self.draw(ctx);
    }

    fn draw(self: *const SomeWidget, ctx: vxfw.DrawContext) std.mem.Allocator.Error!vxfw.Surface {
        // rendering logic
    }
};
```

### Task 1: Add Relevance Scoring to KnowledgeGraph

**File:** `src/graph/graph.zig`

**What:**
Add a method that scores files/modules by relevance to a query string, returning only the top-N most relevant entries.

**New method on `KnowledgeGraph`:**

```zig
/// Score each module/node by relevance to a query using simple keyword matching.
/// Returns sorted slice of (node_id, score) pairs, highest first.
pub fn scoreRelevance(
    self: *KnowledgeGraph,
    allocator: Allocator,
    query: []const u8,
    max_results: u32,
) ![]RelevanceScore {
    var scores = std.ArrayList(RelevanceScore).init(allocator);
    defer scores.deinit(); // only on error path

    const query_lower = try std.ascii.allocLowerString(allocator, query);
    defer allocator.free(query_lower);

    // Split query into words for multi-keyword matching
    var query_words = std.ArrayList([]const u8).init(allocator);
    defer query_words.deinit();
    var word_iter = std.mem.splitScalar(u8, query_lower, ' ');
    while (word_iter.next()) |word| {
        if (word.len > 2) { // skip short words
            try query_words.append(word);
        }
    }

    // Score each node
    var node_iter = self.nodes.iterator();
    while (node_iter.next()) |entry| {
        const node = entry.value_ptr.*;
        var score: f32 = 0.0;

        // Module name match (highest weight)
        const name_lower = std.ascii.allocLowerString(allocator, node.name) catch continue;
        defer allocator.free(name_lower);

        for (query_words.items) |word| {
            if (std.mem.indexOf(u8, name_lower, word)) |_| {
                score += 3.0;
            }
        }

        // File path match
        const path_lower = std.ascii.allocLowerString(allocator, node.file_path) catch continue;
        defer allocator.free(path_lower);

        for (query_words.items) |word| {
            if (std.mem.indexOf(u8, path_lower, word)) |_| {
                score += 2.0;
            }
        }

        // Doc comment match
        if (node.doc_comment) |doc| {
            const doc_lower = std.ascii.allocLowerString(allocator, doc) catch continue;
            defer allocator.free(doc_lower);
            for (query_words.items) |word| {
                if (std.mem.indexOf(u8, doc_lower, word)) |_| {
                    score += 1.5;
                }
            }
        }

        // Signature match
        if (node.signature) |sig| {
            const sig_lower = std.ascii.allocLowerString(allocator, sig) catch continue;
            defer allocator.free(sig_lower);
            for (query_words.items) |word| {
                if (std.mem.indexOf(u8, sig_lower, word)) |_| {
                    score += 1.0;
                }
            }
        }

        if (score > 0.0) {
            try scores.append(.{
                .node_id = node.id,
                .score = score,
            });
        }
    }

    // Sort by score descending
    std.sort.insertion(RelevanceScore, scores.items, {}, relevanceGreaterThan);

    // Return top max_results
    const result_count = @min(scores.items.len, max_results);
    const result = try allocator.alloc(RelevanceScore, result_count);
    @memcpy(result, scores.items[0..result_count]);
    return result;
}

fn relevanceGreaterThan(_: void, a: RelevanceScore, b: RelevanceScore) bool {
    return a.score > b.score;
}
```

**New method: `toRelevantContext`**

```zig
/// Generate compressed context filtered by relevance to a query,
/// limited to a token budget.
pub fn toRelevantContext(
    self: *KnowledgeGraph,
    allocator: Allocator,
    query: []const u8,
    max_tokens: u64,
) ![]const u8 {
    const scores = try self.scoreRelevance(allocator, query, 50);
    defer allocator.free(scores);

    var buf = std.ArrayList(u8).init(allocator);
    defer buf.deinit();
    const writer = buf.writer();

    try writer.print("=== Codebase Context (relevance-filtered) ===\n", .{});
    var tokens_used: u64 = 0;

    for (scores) |score_entry| {
        const node = self.nodes.get(score_entry.node_id) orelse continue;
        const entry_text = try std.fmt.allocPrint(allocator,
            "  {s} [{s}] {s}:{d}",
            .{ node.id, @tagName(node.node_type), node.file_path, node.line }
        );
        defer allocator.free(entry_text);
        const entry_tokens = ContextCompactor.estimateTokens(entry_text);
        if (tokens_used + entry_tokens > max_tokens) break;
        try writer.print("{s}\n", .{entry_text});
        tokens_used += entry_tokens;
    }

    return buf.toOwnedSlice();
}
```

### Task 2: Dynamic Context Source Discovery

**File:** `src/tui/widgets/types.zig`

**What:**
Replace the hardcoded `context_source_files` array with a function that discovers source files dynamically.

**Approach:**
Keep the hardcoded list as a fallback, but add a function that:
1. Walks `src/` directory
2. Collects `.zig` files
3. Filters out test files and generated files
4. Returns a dynamically allocated list

```zig
/// Discover Zig source files in the project for context indexing.
/// Falls back to hardcoded list if directory walk fails.
pub fn discoverSourceFiles(allocator: Allocator) ![]const []const u8 {
    var files = std.ArrayList([]const u8).init(allocator);
    errdefer {
        for (files.items) |f| allocator.free(f);
        files.deinit();
    }

    // Walk src/ directory
    var src_dir = std.fs.cwd().openDir("src", .{ .iterate = true }) catch
        return allocator.dupe([]const u8, &context_source_files);
    defer src_dir.close();

    var walker = src_dir.walk(allocator) catch
        return allocator.dupe([]const u8, &context_source_files);
    defer walker.deinit();

    while (try walker.next()) |entry| {
        if (entry.kind != .file) continue;
        if (!std.mem.endsWith(u8, entry.path, ".zig")) continue;
        if (std.mem.indexOf(u8, entry.path, "test_") != null) continue;

        const full_path = try std.fmt.allocPrint(allocator, "src/{s}", .{entry.path});
        try files.append(full_path);
    }

    // Also add build.zig
    try files.append(try allocator.dupe(u8, "build.zig"));

    return files.toOwnedSlice();
}
```

### Task 3: Token-Aware System Prompt

**File:** `src/tui/chat_tui_app.zig`

**What:**
Modify `refreshEffectiveSystemPrompt` to respect a token budget.

**Constants:**
```zig
const system_prompt_token_budget: u64 = 8000; // max tokens for system prompt context
```

**Changes to `refreshEffectiveSystemPrompt`:**
1. After building the system prompt, estimate its tokens
2. If over budget, truncate the codebase context section
3. Use a simple truncation: keep module overview, drop individual symbols

```zig
fn refreshEffectiveSystemPrompt(self: *Model) !void {
    // ... existing cleanup code ...

    const base_prompt = if (self.system_prompt) |prompt|
        if (prompt.len > 0) prompt else default_system_prompt
    else
        default_system_prompt;

    if (self.codebase_context) |compressed_context| {
        var buf = array_list_compat.ArrayList(u8).init(self.allocator);
        defer buf.deinit();
        const w = buf.writer();

        // Build header
        w.print("{s}\n\n## Codebase Context\n", .{base_prompt}) catch {};

        // Token budget check
        const header_tokens = compaction_mod.ContextCompactor.estimateTokens(base_prompt);
        const tool_section_estimate: u64 = 500; // approximate tokens for tool descriptions
        const remaining_budget = system_prompt_token_budget -| header_tokens -| tool_section_estimate;

        const ctx_tokens = compaction_mod.ContextCompactor.estimateTokens(compressed_context);
        if (ctx_tokens > remaining_budget) {
            // Truncate: keep first remaining_budget*4 chars
            const trunc_len = @min(compressed_context.len, remaining_budget * 4);
            w.print("{s}\n\n[... context truncated to fit budget ...]\n", .{compressed_context[0..trunc_len]}) catch {};
        } else {
            w.print("{s}", .{compressed_context}) catch {};
        }

        // Tool descriptions (always included)
        w.print("\n## Available Tools\n- read_file(path)\n- shell(command)\n- write_file(path, content)\n- glob(pattern)\n- grep(pattern)\n- edit(file_path, old_string, new_string)", .{}) catch {};

        // MCP tools if available
        if (self.mcp_bridge) |bridge| {
            // ... existing MCP tool injection code ...
        }

        self.effective_system_prompt = try buf.toOwnedSlice();
        return;
    }

    // No codebase context — existing path
    // ...
}
```

### Task 4: Relevance-Based Context on Query

**File:** `src/tui/chat_tui_app.zig`

**What:**
Before sending each AI request, rebuild context filtered by the current query.

**Approach:**
In the function that sends the chat request (before calling `sendChatStreamingWithFallback`), extract keywords from the user's latest message and use them to filter context.

However, this is complex to do correctly in the current architecture because:
1. The knowledge graph is built once at init
2. Rebuilding per-query would be slow
3. The graph is freed after initial context build

**Simpler approach for v0.7.0:**
- Store the `KnowledgeGraph` in the Model (don't deinit after `buildCodebaseContext`)
- Add a method `refreshContextForQuery(query)` that:
  1. Calls `kg.toRelevantContext(allocator, query, token_budget)`
  2. Replaces `self.codebase_context`
  3. Calls `refreshEffectiveSystemPrompt()`

**Changes:**
1. Add `knowledge_graph: ?*graph_mod.KnowledgeGraph` to Model
2. In `buildCodebaseContext`: don't `defer kg.deinit()`, store as `self.knowledge_graph`
3. In Model.deinit: clean up knowledge_graph
4. Before sending chat: call `refreshContextForQuery(user_message)`
5. Skip refresh if message is a slash command

**Important:** This is the most complex task in this plan. Consider implementing it AFTER Tasks 1-3 are stable, and only if time permits. The hardcoded full-context dump is acceptable for v0.7.0 if query-based relevance proves too complex.

**Scope gate:** If Tasks 1-3 take more than 80% of the time budget, skip this task. Relevance scoring can ship in v0.8.0.

- **Relevance scoring quality:** Simple keyword matching is basic. Good enough for v0.7.0, real semantic search is v0.8+.
- **Memory:** Storing KnowledgeGraph persistently in Model increases memory. For ~144 files this is negligible.
- **Scope:** Task 4 (query-based refresh) is the riskiest. Scope-gate it — ship Tasks 1-3 first.
